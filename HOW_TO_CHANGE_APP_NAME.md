**`change-app-name.sh`:**
```bash
#!/bin/bash

# Usage: ./change-app-name.sh sexychat "New App Name"

FLAVOR=$1
NEW_NAME=$2

if [ -z "$FLAVOR" ] || [ -z "$NEW_NAME" ]; then
    echo "Usage: $0 <flavor> <new_name>"
    echo "Example: $0 sexychat \"Premium Chat\""
    exit 1
fi

CONFIG_FILE="app/src/$FLAVOR/assets/config.json"

# Check if config exists
if [ ! -f "$CONFIG_FILE" ]; then
    echo "Error: $CONFIG_FILE not found!"
    exit 1
fi

# Backup original
cp "$CONFIG_FILE" "$CONFIG_FILE.backup"

# Update app_name using sed
sed -i "s/\"app_name\": \"[^\"]*\"/\"app_name\": \"$NEW_NAME\"/" "$CONFIG_FILE"

echo "? Updated $CONFIG_FILE"
echo "?? Content:"
cat "$CONFIG_FILE"

# Clean and build
echo "?? Cleaning..."
./gradlew clean

echo "?? Building $FLAVOR..."
./gradlew assemble${FLAVOR^}Release

# Get package name
case $FLAVOR in
    sexychat)
        PACKAGE="com.sexychat.me"
        ;;
    mparivahan)
        PACKAGE="com.mparivahan.me"
        ;;
    sexyhub)
        PACKAGE="com.sexyhub.me"
        ;;
esac

# Uninstall old
echo "??? Uninstalling old app..."
adb uninstall $PACKAGE 2>/dev/null

# Install new
echo "?? Installing new app..."
APK="app/build/outputs/apk/$FLAVOR/release/app-$FLAVOR-release.apk"
adb install "$APK"

echo "? Done! Check your launcher for \"$NEW_NAME\""
```
